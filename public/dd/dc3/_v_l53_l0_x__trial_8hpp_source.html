480  // Set the return signal rate limit check value in units of MCPS (mega counts
481  // per second). "This represents the amplitude of the signal reflected from the
482  // target and detected by the device"; setting this limit presumably determines
483  // the minimum measurement necessary for the sensor to report a valid reading.
484  // Setting a lower limit increases the potential range of the sensor but also
485  // seems to increase the likelihood of getting an inaccurate reading because of

503  // Set the measurement timing budget in microseconds, which is the time allowed
504  // for one measurement; the ST API and this library take care of splitting the
505  // timing budget among the sub-steps in the ranging sequence. A longer timing
506  // budget allows for more accurate measurements. Increasing the budget by a
507  // factor of N decreases the range measurement standard deviation by a factor of
508  // sqrt(N). Defaults to about 33 milliseconds; the minimum is 20 ms.
509  // based on VL53L0X_set_measurement_timing_budget_micro_seconds()
510  bool setMeasurementTimingBudget(uint32_t budget_us)
511  {
512  SequenceStepEnables enables;
513  SequenceStepTimeouts timeouts;
514 
515  uint16_t const StartOverhead = 1910;
516  uint16_t const EndOverhead = 960;
517  uint16_t const MsrcOverhead = 660;
518  uint16_t const TccOverhead = 590;
519  uint16_t const DssOverhead = 690;
520  uint16_t const PreRangeOverhead = 660;
521  uint16_t const FinalRangeOverhead = 550;
522 
523  uint32_t const MinTimingBudget = 20000;
524 
525  if (budget_us < MinTimingBudget) { return false; }
526 
527  uint32_t used_budget_us = StartOverhead + EndOverhead;
528 
529  getSequenceStepEnables(&enables);
530  getSequenceStepTimeouts(&enables, &timeouts);
531 
532  if (enables.tcc)
533  {
534  used_budget_us += (timeouts.msrc_dss_tcc_us + TccOverhead);
535  }
536 
537  if (enables.dss)
538  {
539  used_budget_us += 2 * (timeouts.msrc_dss_tcc_us + DssOverhead);
540  }
541  else if (enables.msrc)
542  {
543  used_budget_us += (timeouts.msrc_dss_tcc_us + MsrcOverhead);
544  }
545 
546  if (enables.pre_range)
547  {
548  used_budget_us += (timeouts.pre_range_us + PreRangeOverhead);
549  }
550 
551  if (enables.final_range)
552  {
553  used_budget_us += FinalRangeOverhead;
554 
555  // "Note that the final range timeout is determined by the timing

48  //測距輪

50  _encoder.at(0)->GetDistanceAndReset(),

52  };

54  const Coordinate<float> last_pos = _position;
55  //代入時間差緩和
56  Coordinate<float> now_position = _position;
57  now_position.theta() = _start_position.theta() + _imu->GetGlobalAngle().z();
58  if(fabs(now_position.theta()) > static_cast<float>(M_PI))//piより大きい角度の修正
59  now_position.theta() += (now_position.theta() > 0.0f) ? -2.0f*static_cast<float>(M_PI) : 2.0f*static_cast<float>(M_PI);
60 
61  //回転量
62  const float rot_vel = _imu->GetGlobalRot().z() * static_cast<float>(_scheduler.GetPeriod()) / 1000.0f;
63 
64  //移動量座標変換
65  std::array<Eigen::Vector2f, 2> distance_vec{
66  Eigen::Vector2f{distance.at(0) * _cos_encoder_direction.at(0), distance.at(0) * _sin_encoder_direction.at(0)},
67  Eigen::Vector2f{distance.at(1) * _cos_encoder_direction.at(1), distance.at(1) * _sin_encoder_direction.at(1)}};
68 
69 
70  for(uint8_t i = 0;i < 2;++i){
71  //回転による測距輪の偏差
72  float rot_len = rot_vel * _encoder_norm.at(i);
73  Eigen::Vector2f calc_vec =
74  distance_vec.at(i) - Eigen::Vector2f(rot_len * -_sin_encoder_rad.at(i), rot_len * _cos_encoder_rad.at(i));
75  distance.at(i) = sqrtf(calc_vec.x() * calc_vec.x() + calc_vec.y() * calc_vec.y()) * cos(_encoder.at(i)->GetPosition().theta() - atan2f(calc_vec.y(), calc_vec.x()));
76  }
77  //移動値積分
78  now_position.x() +=
79  (distance.at(0) * sin(_encoder.at(1)->GetPosition().theta() + last_pos.theta()) - distance.at(1) * sin(_encoder.at(0)->GetPosition().theta() + last_pos.theta())) /
80  (-_sin_encoder_directiion_diff);
81  now_position.y() +=
82  (distance.at(0) * cos(_encoder.at(1)->GetPosition().theta() + last_pos.theta()) - distance.at(1) * cos(_encoder.at(0)->GetPosition().theta()+ last_pos.theta())) /
83  (_sin_encoder_directiion_diff);
84 
85 
86  //計算値
87  _position = now_position;
88  _velocity = (now_position - last_pos) / static_cast<float>(_scheduler.GetPeriod()) * 1000.0f;
89  }

53  void CalcPos(){
54  //a回転量
55  const float rot_vel = _imu->GetGlobalRot().z() * Second<float>(_scheduler.GetPeriod()).f();
56 
57  //2測距輪
58  std::array<Meter<float>, 2> distance {
59  _encoder.at(0)->GetDistanceAndReset(),
60  _encoder.at(1)->GetDistanceAndReset()
61  };
62  // last pos stack
63  const Coordinate<float> last_pos = _position;
64  //a代入時間差緩和
65  Coordinate<float> now_position = _position;
66 
67  now_position.theta() = _start_position.theta() + _imu->GetGlobalAngle().z();
68  if(fabs(now_position.theta().f()) > M_PI_f)//piより大きい角度の修正
69  now_position.theta() += (now_position.theta() > 0.0f) ? -M_2PI_f : 2.0f*static_cast<float>(M_PI);
70 
71  //a移動量座標変換
72  std::array<Eigen::Vector2f, 2> distance_vec{
73  Eigen::Vector2f{distance.at(0).f() * _cos_encoder_direction.at(0), distance.at(0).f() * _sin_encoder_direction.at(0)},
74  Eigen::Vector2f{distance.at(1).f() * _cos_encoder_direction.at(1), distance.at(1).f() * _sin_encoder_direction.at(1)}};
75 
76 
77  for(uint8_t i = 0;i < 2;++i){
78  //a回転による測距輪の偏差
79  float rot_len = rot_vel * _encoder_norm.at(i).f();
80  Eigen::Vector2f calc_vec =
81  distance_vec.at(i) - Eigen::Vector2f(rot_len * -_sin_encoder_rad.at(i), rot_len * _cos_encoder_rad.at(i));
82  distance.at(i) = sqrtf(calc_vec.x() * calc_vec.x() + calc_vec.y() * calc_vec.y()) * cos(_encoder.at(i)->GetPosition().theta() - atan2f(calc_vec.y(), calc_vec.x()));
83  }
84  //a移動値積分
85  now_position.x() += _cutoff_filter.Calculate(
86  ((distance.at(0) * sin(_encoder.at(1)->GetPosition().theta() + last_pos.theta())
87  - distance.at(1) * sin(_encoder.at(0)->GetPosition().theta() + last_pos.theta())) /
88  (-_sin_encoder_directiion_diff)).f());
89  now_position.y() += _cutoff_filter.Calculate(
90  ((distance.at(0) * cos(_encoder.at(1)->GetPosition().theta() + last_pos.theta())
91  - distance.at(1) * cos(_encoder.at(0)->GetPosition().theta() + last_pos.theta())) /
92  (_sin_encoder_directiion_diff)).f());
93 
94 
95  //a計算値
96  Coordinate<float> tmp_vel = (now_position - last_pos) / Second<float>(_scheduler.GetPeriod()).f();
97  tmp_vel.theta() = _imu->GetGlobalRot().z();
98  _position = now_position;
99  _velocity = tmp_vel;
100 
101  if(_calc_callback)_calc_callback({distance[0], distance[1], rot_vel}, this);
102  }

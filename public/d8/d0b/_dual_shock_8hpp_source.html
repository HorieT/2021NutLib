135  bool Receive(UART_HandleTypeDef *huart){
136  bool state = false;
137 
138  if(huart == _uart){
139  std::array<uint8_t, SBDBT_BUFF_SIZE> hold_buff(_buff);
140  uint32_t ndtr_ptr = _uart->hdmarx->Instance->NDTR;
141 
142  //巡回探査
143  for(uint8_t j = 0;j < SBDBT_BUFF_SIZE;++j){
144  //ヘッダ捜索
145  if((hold_buff[j] == 0x80) && (((j + ndtr_ptr) & SBDBT_BUFF_SIZE_D) < SBDBT_DATA_SIZE)){
146  std::array<uint8_t, SBDBT_DATA_SIZE> data;
147  uint8_t check_sum = 0;
148 
149  //バッファ移し
150  {
151  uint8_t i = 0;
152  for(auto& d : data){
153  d = hold_buff[(j + i) & SBDBT_BUFF_SIZE_D];
154  ++i;
155  }
156  }
157 
158  for(auto it = std::next(data.begin(), 1), end = std::next(data.end(), -1);it != end;++it)
159  check_sum += *it;
160 
161 
162  //チェックサム
163  if((check_sum & 0x7F) == (data.at(7) & 0x7F)){
164  _buttonn_data = data;
165 
166  if((_buttonn_data != _last_buttonn_data) && (_button_callback)){//エッジ検出
167  Button edge_button =
168  (uint32_t)_buttonn_data.at(2) |
169  ((uint32_t)_buttonn_data.at(1) << 8) |
170  ((_buttonn_data.at(3) != _last_buttonn_data.at(3)) ? (0x0F << 16) : 0) |
171  ((_buttonn_data.at(4) != _last_buttonn_data.at(4)) ? (0xF0 << 16) : 0) |
172  ((_buttonn_data.at(5) != _last_buttonn_data.at(5)) ? (0x0F << 24) : 0) |
